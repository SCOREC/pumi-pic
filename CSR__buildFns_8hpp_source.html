33 lid_t given_particles = particle_elements.size();

36 // create a pointer to the offsets array that we can access in a kokkos parallel_for
37 auto offset_cpy = offsets;
38 kkLidView particle_indices(Kokkos::ViewAllocateWithoutInitializing("particle_indices"), num_ptcls);
39 // SS3 insert code to set the entries of particle_indices>
40 kkLidView row_indices(Kokkos::ViewAllocateWithoutInitializing("row_indices"), num_elems+1);
41 Kokkos::deep_copy(row_indices, offset_cpy);
42
43 // atomic_fetch_add to increment from the beginning of each element
44 // when filling (offset[element] is start of element)

47 };
