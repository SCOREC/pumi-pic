229template<class DataTypes, typename MemSpace>
230void SellCSigma<DataTypes, MemSpace>::construct(kkLidView ptcls_per_elem,
231 kkGidView element_gids,
232 kkLidView particle_elements,
233 MTVs particle_info,
234 MPI_Comm mpi_comm) {
235 Kokkos::Profiling::pushRegion("scs_construction");
236 tryShuffling = true;
237 int comm_size;
238 MPI_Comm_size(mpi_comm, &comm_size);
239 int comm_rank;
240 MPI_Comm_rank(mpi_comm, &comm_rank);
241
242 C_max = policy.team_size();
243 C_ = chooseChunkHeight(C_max, ptcls_per_elem);
244
245 if(!comm_rank)
246 printInfo( "Building SCS with C: %d sigma: %d V: %d\n",C_,sigma,V_);
247 //Perform sorting
248 kkLidView ptcls;
249 kkLidView index;
250 sigmaSort(ptcls, index, num_elems,ptcls_per_elem, sigma);
251
252 // Number of chunks without vertical slicing
253 kkLidView chunk_widths;
254 constructChunks(ptcls, index, num_chunks, chunk_widths, row_to_element, element_to_row);
255 num_rows = num_chunks * C_;
256
257 if (element_gids.size() > 0) {
258 createGlobalMapping(element_gids, element_to_gid, element_gid_to_lid);
259 }
260
261 //Create offsets into each chunk/vertical slice
262 constructOffsets(num_chunks, num_slices, chunk_widths, offsets, slice_to_chunk,capacity_);
263
264 //Allocate the SCS and backup with extra space
265 lid_t cap = capacity_;
266 particle_mask = Kokkos::View<bool*>(Kokkos::ViewAllocateWithoutInitializing("particle_mask"), cap);
267 if (extra_padding > 0)
268 cap *= (1 + extra_padding);
269 CreateViews<device_type, DataTypes>(ptcl_data, cap);
270 if (!always_realloc)
271 CreateViews<device_type, DataTypes>(scs_data_swap, cap);
272 swap_size = current_size = cap;
273
274 if (num_ptcls > 0) {
275 kkLidView chunk_starts;
276 setupParticleMask(particle_mask, ptcls, chunk_widths, chunk_starts);
277
278 //If particle info is provided then enter the information
279 lid_t given_particles = particle_elements.size();
280 if (given_particles > 0 && particle_info != NULL) {
281 initSCSData(chunk_starts, particle_elements, particle_info);
282 }
283 }
284 Kokkos::Profiling::popRegion();
285}

337template <class MSpace>
338typename SellCSigma<DataTypes, MemSpace>::template Mirror<MSpace>* SellCSigma<DataTypes, MemSpace>::copy() {
339 if (std::is_same<memory_space, typename MSpace::memory_space>::value) {
340 printError("Copy to same memory space not supported\n");
341 exit(EXIT_FAILURE);
342 }
343 const auto cmax = maxChunk<MSpace>(C_max);
344 Mirror<MSpace>* mirror_copy = new SellCSigma<DataTypes, MSpace>(cmax);
345 //Call Particle structures copy
346 mirror_copy->copy(this);
347 //Copy constants
348 mirror_copy->C_ = C_;
349 mirror_copy->C_max = C_max;
350 mirror_copy->V_ = V_;
351 mirror_copy->sigma = sigma;
352 mirror_copy->num_chunks = num_chunks;
353 mirror_copy->num_slices = num_slices;
354 mirror_copy->current_size = current_size;
355 mirror_copy->swap_size = swap_size;
356 mirror_copy->extra_padding = extra_padding;
357 mirror_copy->shuffle_padding = shuffle_padding;
358 mirror_copy->pad_strat = pad_strat;
359 mirror_copy->minimize_size = minimize_size;
360 mirror_copy->always_realloc = always_realloc;
361 mirror_copy->tryShuffling = tryShuffling;
362 mirror_copy->num_empty_elements = num_empty_elements;
363
364 //Create the swap space
365 mirror_copy->scs_data_swap = createMemberViews<DataTypes, memory_space>(swap_size);
366 //Deep copy each view
367 mirror_copy->slice_to_chunk = typename Mirror<MSpace>::kkLidView("mirror slice_to_chunk",
368 slice_to_chunk.size());
369 Kokkos::deep_copy(mirror_copy->slice_to_chunk, slice_to_chunk);
370 mirror_copy->particle_mask = Kokkos::View<bool*,MSpace>("mirror particle_mask",
371 particle_mask.size());
372 Kokkos::deep_copy(mirror_copy->particle_mask, particle_mask);
373 mirror_copy->offsets = typename Mirror<MSpace>::kkLidView("mirror offsets", offsets.size());
374 Kokkos::deep_copy(mirror_copy->offsets, offsets);
375 mirror_copy->row_to_element = typename Mirror<MSpace>::kkLidView("mirror row_to_element",
376 row_to_element.size());
377 Kokkos::deep_copy(mirror_copy->row_to_element, row_to_element);
378 mirror_copy->element_to_row = typename Mirror<MSpace>::kkLidView("mirror element_to_row",
379 element_to_row.size());
380 Kokkos::deep_copy(mirror_copy->element_to_row, element_to_row);
381 mirror_copy->element_to_gid = typename Mirror<MSpace>::kkGidView("mirror element_to_gid",
382 element_to_gid.size());
383 Kokkos::deep_copy(mirror_copy->element_to_gid, element_to_gid);
384 //Deep copy the gid mapping
385 mirror_copy->element_gid_to_lid.create_copy_view(element_gid_to_lid);
386 return mirror_copy;
387}

403template<class DataTypes, typename MemSpace>
404void SellCSigma<DataTypes,MemSpace>::printFormat(const char* prefix) const {
405 //Transfer everything to the host
406 kkLidHostMirror slice_to_chunk_host = deviceToHost(slice_to_chunk);
407 kkGidHostMirror element_to_gid_host = deviceToHost(element_to_gid);
408 kkLidHostMirror row_to_element_host = deviceToHost(row_to_element);
409 kkLidHostMirror offsets_host = deviceToHost(offsets);
410 Kokkos::View<bool*>::host_mirror_type particle_mask_host = deviceToHost(particle_mask);
411
412 std::stringstream ss;
413 char buffer[1000];
414 char* ptr = buffer;
415 int num_chars;
416
417 num_chars = sprintf(ptr, "%s\n", prefix);
418 num_chars += sprintf(ptr+num_chars,"Particle Structures Sell-C-Sigma C: %d sigma: %d V: %d.\n", C_, sigma, V_);
419 num_chars += sprintf(ptr+num_chars,"Number of Elements: %d.\nNumber of Particles: %d.\n", num_elems, num_ptcls);
420 num_chars += sprintf(ptr+num_chars,"Number of Chunks: %d.\nNumber of Slices: %d.\n", num_chunks, num_slices);
421 buffer[num_chars] = '\0';
422 ss << buffer;
423
424 lid_t last_chunk = -1;
425 for (lid_t i = 0; i < num_slices; ++i) {
426 lid_t chunk = slice_to_chunk_host(i);
427 if (chunk != last_chunk) {
428 last_chunk = chunk;
429 num_chars = sprintf(ptr," Chunk %d. Elements", chunk);
430 if (element_to_gid_host.size() > 0)
431 num_chars += sprintf(ptr+num_chars,"(GID)");
432 num_chars += sprintf(ptr+num_chars,":");
433 buffer[num_chars] = '\0';
434 ss << buffer;
435
436 for (lid_t row = chunk*C_; row < (chunk+1)*C_; ++row) {
437 lid_t elem = row_to_element_host(row);
438 num_chars = sprintf(ptr," %d", elem);
439 if (element_to_gid_host.size() > 0)
440 num_chars += sprintf(ptr+num_chars,"(%ld)", element_to_gid_host(elem));
441 buffer[num_chars] = '\0';
442 ss << buffer;
443 }
444 num_chars = sprintf(ptr,"\n");
445 buffer[num_chars] = '\0';
446 ss << buffer;
447 }
448 num_chars = sprintf(ptr," Slice %d", i);
449 buffer[num_chars] = '\0';
450 ss << buffer;
451 for (lid_t j = offsets_host(i); j < offsets_host(i+1); ++j) {
452 if ((j - offsets_host(i)) % C_ == 0)
453 num_chars = sprintf(ptr," |");
454 num_chars += sprintf(ptr+num_chars," %d", particle_mask_host(j));
455 buffer[num_chars] = '\0';
456 ss << buffer;
457 }
458 num_chars = sprintf(ptr,"\n");
459 buffer[num_chars] = '\0';
460 ss << buffer;
461 }
462 std::cout << ss.str();
463}

465template <class DataTypes, typename MemSpace>
466void SellCSigma<DataTypes, MemSpace>::printMetrics(MPI_Comm mpi_comm) const {
467
468 //Gather metrics
469 kkLidView padded_cells("padded_cells", 1);
470 kkLidView padded_slices("padded_slices", 1);
471 const lid_t league_size = num_slices;
472 const lid_t team_size = C_;
473 const PolicyType policy(league_size, team_size);
474 auto offsets_cpy = offsets;
475 auto slice_to_chunk_cpy = slice_to_chunk;
476 auto row_to_element_cpy = row_to_element;
477 auto particle_mask_cpy = particle_mask;
478 Kokkos::parallel_for("GatherMetrics", policy,
479 KOKKOS_LAMBDA(const typename PolicyType::member_type& thread) {
480 const lid_t slice = thread.league_rank();
481 const lid_t slice_row = thread.team_rank();
482 const lid_t rowLen = (offsets_cpy(slice+1)-offsets_cpy(slice))/team_size;
483 const lid_t start = offsets_cpy(slice) + slice_row;
484 const lid_t row = slice_to_chunk_cpy(slice) * team_size + slice_row;
485 const lid_t element_id = row_to_element_cpy(row);
486 lid_t np = 0;
487 for (lid_t p = 0; p < rowLen; ++p) {
488 const lid_t particle_id = start+(p*team_size);
489 const bool mask = particle_mask_cpy(particle_id);
490 np += !mask;
491 }
492 Kokkos::atomic_fetch_add(&padded_cells[0],np);
493 thread.team_reduce(Kokkos::Sum<lid_t, MemSpace>(np));
494 if (slice_row == 0)
495 Kokkos::atomic_fetch_add(&padded_slices[0], np > 0);
496 });
497
498 lid_t num_padded = getLastValue(padded_cells);
499 lid_t num_padded_slices = getLastValue(padded_slices);
500
501 int comm_rank;
502 MPI_Comm_rank(mpi_comm, &comm_rank);
503
504 char buffer[1000];
505 char* ptr = buffer;
506
507 //Header
508 ptr += sprintf(ptr, "Metrics %d, C %d, V %d, sigma %d\n", comm_rank, C_, V_, sigma);
509 //Sizes
510 ptr += sprintf(ptr, "Nelems %d, Nchunks %d, Nslices %d, Nptcls %d, Capacity %d, "
511 "Allocation %lu\n", nElems(), num_chunks, num_slices, nPtcls(),
512 capacity(), current_size + swap_size);
513 //Padded Cells
514 ptr += sprintf(ptr, "Padded Cells <Tot %%> %d %.3f\n", num_padded,
515 num_padded * 100.0 / particle_mask.size());
516 //Padded Slices
517 ptr += sprintf(ptr, "Padded Slices <Tot %%> %d %.3f\n", num_padded_slices,
518 num_padded_slices * 100.0 / num_slices);
519 //Empty Elements
520 ptr += sprintf(ptr, "Empty Rows <Tot %%> %d %.3f\n", num_empty_elements,
521 num_empty_elements * 100.0 / numRows());
522
523 printInfo("%s\n",buffer);
524}

526template <class DataTypes, typename MemSpace>
527template <typename FunctionType>
528void SellCSigma<DataTypes, MemSpace>::parallel_for(FunctionType& fn, std::string name) {
529 if (nPtcls() == 0)
530 return;
531 FunctionType* fn_d = gpuMemcpy(fn);
532 const lid_t league_size = num_slices;
533 const lid_t team_size = C_;
534 const PolicyType policy(league_size, team_size);
535 auto offsets_cpy = offsets;
536 auto slice_to_chunk_cpy = slice_to_chunk;
537 auto row_to_element_cpy = row_to_element;
538 auto particle_mask_cpy = particle_mask;
539 Kokkos::parallel_for(name, policy,
540 KOKKOS_LAMBDA(const typename PolicyType::member_type& thread) {
541 const lid_t slice = thread.league_rank();
542 const lid_t slice_row = thread.team_rank();
543 const lid_t rowLen = (offsets_cpy(slice+1)-offsets_cpy(slice))/team_size;
544 const lid_t start = offsets_cpy(slice) + slice_row;
545 Kokkos::parallel_for(Kokkos::TeamThreadRange(thread, team_size), [=] (lid_t& j) {
546 const lid_t row = slice_to_chunk_cpy(slice) * team_size + slice_row;
547 const lid_t element_id = row_to_element_cpy(row);
548 Kokkos::parallel_for(Kokkos::ThreadVectorRange(thread, rowLen), [=] (lid_t& p) {
549 const lid_t particle_id = start+(p*team_size);
550 const bool mask = particle_mask_cpy(particle_id);
551 (*fn_d)(element_id, particle_id, mask);
552 });
553 });
554 });
555#ifdef PP_USE_GPU
556 gpuFree(fn_d);
557#endif
558}
