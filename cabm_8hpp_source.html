145 lid_t num_elements, lid_t num_particles,
146 kkLidView particles_per_element,
147 kkGidView element_gids,
148 kkLidView particle_elements, // optional
149 MTVs particle_info, // optional
150 MPI_Comm mpi_comm) : // optional
151 ParticleStructure<DataTypes, MemSpace>(),
152 policy(p),
153 element_gid_to_lid(num_elements),
154 extra_padding(0.05) // default extra padding at 5%
155 {
156 assert(num_elements == particles_per_element.size());
157 num_elems = num_elements;
158 num_rows = num_elems;
159 num_ptcls = num_particles;
160 int comm_rank;
161 MPI_Comm_rank(mpi_comm, &comm_rank);
162 if(!comm_rank)
163 printInfo( "building CabM\n");
164
165 // build view of offsets for SoA indices within particle elements
166 offsets = buildOffset(particles_per_element, num_ptcls, extra_padding, padding_start);
167 // set num_soa_ from the last entry of offsets
168 num_soa_ = getLastValue(offsets);
169 // calculate capacity_ from num_soa_ and max size of an SoA
170 capacity_ = num_soa_*AoSoA_t::vector_length;
171 // initialize appropriately-sized AoSoA and copy for swapping
172 aosoa_ = makeAoSoA(capacity_, num_soa_);
173 aosoa_swap = makeAoSoA(capacity_, num_soa_);
174 // get array of parents element indices for particles
175 parentElms_ = getParentElms(num_elems, num_soa_, offsets);
176 // set active mask
177 setActive(particles_per_element);
178 // get global ids
179 if (element_gids.size() > 0) {
180 createGlobalMapping(element_gids, element_to_gid, element_gid_to_lid);
181 }
182 // populate AoSoA with input data if given
183 if (particle_elements.size() > 0 && particle_info != NULL) {
184 if(!comm_rank) printInfo( "initializing CabM data\n");
185 fillAoSoA(particle_elements, particle_info); // initialize data
186 }

247 if (nPtcls() == 0)

250 // move function pointer to GPU (if needed)

252 kkLidView parentElms_cpy = parentElms_;
253 const auto soa_len = AoSoA_t::vector_length;
254 const auto mask = Cabana::slice<CM_DT::size-1>(*aosoa_); // get active mask

258 const lid_t elm = parentElms_cpy(soa); // calculate element
259 const lid_t particle_id = soa*soa_len + ptcl; // calculate overall index
260 (*fn_d)(elm, particle_id, mask.access(soa,ptcl));
261 }, name);

269 void CabM<DataTypes, MemSpace>::printMetrics(MPI_Comm mpi_comm) const {
270 // Sum number of empty cells
271 auto mask = Cabana::slice<CM_DT::size-1>(*aosoa_);
272 kkLidView padded_cells("num_padded_cells",1);
273 Kokkos::parallel_for("count_padding", capacity_,
274 KOKKOS_LAMBDA(const lid_t ptcl_id) {
275 Kokkos::atomic_fetch_add(&padded_cells(0), !mask(ptcl_id));
276 });
277 // Sum number of empty elements
278 kkLidHostMirror offsets_host = deviceToHost(offsets);
279 lid_t num_empty_elements = 0;
280 if (num_soa_ == 0)
281 num_empty_elements = num_elems;
282 else {
283 for (int i = 0; i < num_elems; i++) {
284 if (i != 0 && (offsets_host(i) == offsets_host(i-1)) )
285 num_empty_elements++;
286 }
287 }
288
289 lid_t num_padded = getLastValue(padded_cells);
290
291 int comm_rank;
292 MPI_Comm_rank(mpi_comm, &comm_rank);
293
294 char buffer[1000];
295 char* ptr = buffer;
296
297 // Header
298 ptr += sprintf(ptr, "Metrics (Rank %d)\n", comm_rank);
299 // Sizes
300 ptr += sprintf(ptr, "Number of Elements %d, Number of SoA %d, Number of Particles %d, Capacity %d\n",
301 num_elems, num_soa_, num_ptcls, capacity_);
302 // Padded Cells
303 ptr += sprintf(ptr, "Padded Cells <Tot %%> %d %.3f%%\n", num_padded,
304 num_padded * 100.0 / capacity_);
305 // Empty Elements
306 ptr += sprintf(ptr, "Empty Elements <Tot %%> %d %.3f%%\n", num_empty_elements,
307 num_empty_elements * 100.0 / num_elems);
308
309 printInfo("%s\n", buffer);
310 }

312 template <class DataTypes, typename MemSpace>
313 void CabM<DataTypes, MemSpace>::printFormat(const char* prefix) const {
314 kkGidHostMirror element_to_gid_host = deviceToHost(element_to_gid);
315 kkLidHostMirror offsets_host = deviceToHost(offsets);
316 kkLidHostMirror parents_host = deviceToHost(parentElms_);
317 const auto soa_len = AoSoA_t::vector_length;
318
319 kkLidView mask(Kokkos::ViewAllocateWithoutInitializing("offsets_host"), capacity_);
320 auto mask_slice = Cabana::slice<CM_DT::size-1>(*aosoa_);
321 Kokkos::parallel_for("copy_mask", capacity_,
322 KOKKOS_LAMBDA(const lid_t ptcl_id) {
323 mask(ptcl_id) = mask_slice(ptcl_id);
324 });
325 kkLidHostMirror mask_host = deviceToHost(mask);
326
327 std::stringstream ss;
328 char buffer[1000];
329 char* ptr = buffer;
330 int num_chars;
331
332 num_chars = sprintf(ptr, "%s\n", prefix);
333 num_chars += sprintf(ptr+num_chars,"Particle Structures CabM\n");
334 num_chars += sprintf(ptr+num_chars,"Number of Elements: %d.\nNumber of SoA: %d.\nNumber of Particles: %d.", num_elems, num_soa_, num_ptcls);
335 buffer[num_chars] = '\0';
336 ss << buffer;
337
338 lid_t last_soa = -1;
339 lid_t last_elm = -1;
340 for (int i = 0; i < capacity_; i++) {
341 lid_t soa = i / soa_len;
342 lid_t elm = parents_host(soa);
343 if (last_soa == soa) {
344 num_chars = sprintf(ptr," %d", mask_host(i));
345 buffer[num_chars] = '\0';
346 ss << buffer;
347 }
348 else {
349 if (element_to_gid_host.size() > 0) {
350 if (last_elm != elm)
351 num_chars = sprintf(ptr,"\n Element %2d(%2ld) | %d", elm, element_to_gid_host(elm), mask_host(i));
352 else
353 num_chars = sprintf(ptr,"\n | %d", mask_host(i));
354 buffer[num_chars] = '\0';
355 ss << buffer;
356 }
357 else {
358 if (last_elm != elm)
359 num_chars = sprintf(ptr,"\n Element %2d | %d", elm, mask_host(i));
360 else
361 num_chars = sprintf(ptr,"\n | %d", mask_host(i));
362 buffer[num_chars] = '\0';
363 ss << buffer;
364 }
365 }
366
367 last_soa = soa;
368 last_elm = elm;
369 }
370 ss << "\n";
371 std::cout << ss.str();
372 }

377 if (std::is_same<memory_space, typename MSpace::memory_space>::value) {
378 printError("Copy to same memory space not supported\n");
379 exit(EXIT_FAILURE);
380 }
381 Mirror<MSpace>* mirror_copy = new CabM<DataTypes, MSpace>();
382 //Call Particle structures copy
383 mirror_copy->copy(this);
384 //Copy constants
385 mirror_copy->num_soa_ = num_soa_;
386 mirror_copy->padding_start = padding_start;
387 mirror_copy->extra_padding = extra_padding;
388
389 //copy AoSoA
390 mirror_copy->aosoa_ = new typename CabM<DataTypes, MSpace>::AoSoA_t(std::string(aosoa_->label()).append("_mirror"), aosoa_->size());
391 Cabana::deep_copy(*(mirror_copy->aosoa_), *aosoa_);
392 //Create the swap space
393 mirror_copy->aosoa_swap = new typename CabM<DataTypes, MSpace>::AoSoA_t(std::string(aosoa_swap->label()).append("_mirror"), aosoa_swap->size());
394 Cabana::deep_copy(*(mirror_copy->aosoa_swap), *aosoa_swap);
395
396 //Deep copy each view
397 mirror_copy->parentElms_ = typename Mirror<MSpace>::kkLidView("mirror parentElms_",
398 parentElms_.size());
399 Kokkos::deep_copy(mirror_copy->parentElms_, parentElms_);
400 mirror_copy->offsets = typename Mirror<MSpace>::kkLidView("mirror offsets", offsets.size());
401 Kokkos::deep_copy(mirror_copy->offsets, offsets);
402 mirror_copy->element_to_gid = typename Mirror<MSpace>::kkGidView("mirror element_to_gid",
403 element_to_gid.size());
404 Kokkos::deep_copy(mirror_copy->element_to_gid, element_to_gid);
405 //Deep copy the gid mapping
406 mirror_copy->element_gid_to_lid.create_copy_view(element_gid_to_lid);
407 return mirror_copy;
408 }
