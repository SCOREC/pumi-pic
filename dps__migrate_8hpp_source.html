36 // Count number of particles to send to each process
37 kkLidView num_send_particles("num_send_particles", comm_size + 1);

39 const lid_t process = new_process(particle_id);

50 MPI_Request* count_send_requests = NULL;
51 if (num_send_ranks > 0)

55 if (dist.isWorld())

57 else {

62 PS_Comm_Isend(num_send_particles, i, 1, rank, 0, dist.mpi_comm(),
63 count_send_requests + request_index);
64 PS_Comm_Irecv(num_recv_particles, i, 1, rank, 0, dist.mpi_comm(),
65 count_recv_requests + request_index);

67 }
68 }
