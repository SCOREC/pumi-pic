25o::LOs searchAndBuildMap(o::Mesh* mesh, o::Reals element_centroids,
26 o::Reals projected_points, o::LOs starting_element) {
27 o::LO num_points = starting_element.size();
28
29 //Create PS for the projected points to perform adjacency search on
30 PSpt::kkLidView ptcls_per_elem("ptcls_per_elem", mesh->nelems());
31 PSpt::kkLidView point_element("point_element", num_points);
32 auto point_info = particle_structs::createMemberViews<Point>(num_points);
33 auto start_pos = particle_structs::getMemberView<Point, 0>(point_info);
34 auto end_pos = particle_structs::getMemberView<Point, 1>(point_info);
35 auto point_id = particle_structs::getMemberView<Point, 2>(point_info);
36 auto countPointsInElement = OMEGA_H_LAMBDA(const o::LO& id) {
37 const o::LO elm = starting_element[id];
38 Kokkos::atomic_fetch_add(&(ptcls_per_elem(elm)), 1); //can just be 'add', no fetch needed
39 point_id(id) = id;
40 point_element(id) = elm;
41 for (int i = 0; i < 2; ++i) {
42 start_pos(id,i) = element_centroids[elm*2+i];
43 end_pos(id,i) = projected_points[id*2+i];
44 }
45 };
46 o::parallel_for(num_points, countPointsInElement, "countPointsInElement");
47
48 const int sigma = INT_MAX;
49 const int V = 64;
50 Kokkos::TeamPolicy<Kokkos::DefaultExecutionSpace> policy = pumipic::TeamPolicyAuto(10000, 32);
51 PSpt::kkGidView empty_gids("empty_gids", 0);
52 PSpt* gyro_ps = new ps::SellCSigma<Point>(policy, sigma, V,
53 mesh->nelems(), num_points,
54 ptcls_per_elem, empty_gids,
55 point_element, point_info);
56 printf("created ps for gyro mapping with %d points and %d elms\n",
57 num_points, mesh->nelems());
58
59 //Adjacency search
60 auto start = gyro_ps->get<0>();
61 auto end = gyro_ps->get<1>();
62 auto pids = gyro_ps->get<2>();
63 int maxLoops = 100;
64 int psCapacity = gyro_ps->capacity();
65 o::Write<o::LO> elem_ids(psCapacity, -1);
66 bool isFound = p::search_mesh_2d(*mesh, gyro_ps, start, end, pids,
67 elem_ids, maxLoops);
68 assert(isFound);
69
70 const auto numElms = mesh->nelems();
71 //Gyro avg mapping: 3 vertices per ring point (Assumes all elements are triangles)
72 const o::LO nvpe = 3;
73 o::Write<o::LO> gyro_avg_map(nvpe * num_points, -1);
74 auto elm2Verts = mesh->ask_down(mesh->dim(), 0);
75 auto createGyroMapping = PS_LAMBDA(const int&, const int& pid, const int& mask) {
76 const o::LO parent = elem_ids[pid];
77 if (mask && parent >= 0) { //skip points outside the domain (parent == -1)
78 assert(parent>=0 && parent<numElms);
79 const o::LO id = pids(pid);
80 const o::LO start_index = id* nvpe;
81 const o::LO start_elm = parent*nvpe;
82 for (int i = 0; i < 3; ++i)
83 gyro_avg_map[start_index+i] = elm2Verts.ab2b[start_elm+i];
84 }
85 };
86 ps::parallel_for(gyro_ps, createGyroMapping);
87 delete gyro_ps;
88 particle_structs::destroyViews<Point>(point_info);
89 return o::LOs(gyro_avg_map);
90}

95/* Build gyro-avg mapping */
96void createGyroRingMappings(o::Mesh* mesh, o::LOs& forward_map,
97 o::LOs& backward_map) {
98 Kokkos::Profiling::pushRegion("xgcm_createGyroRingMappings");
99 const auto gr = gyro_rmax;
100 const auto gnr = gyro_num_rings;
101 const auto gppr = gyro_points_per_ring;
102 const auto gt = gyro_theta; //degrees to offset ring point distribution
103 o::LO nverts = mesh->nverts();
104 o::LO num_points = nverts * gnr * gppr;
105 o::Write<o::Real> ring_points(num_points * 2);
106 o::Reals coords = mesh->coords();
107 const auto torad = M_PI/180;
108
109 //Calculate points of each ring around every vertex
110 auto generateRingPoints = OMEGA_H_LAMBDA(const o::LO& id) {
111 const o::LO point_id = id % gppr;
112 const o::LO id2 = id /gppr;
113 const o::LO ring_id = id2 % gnr;
114 const o::LO vert_id = id2 / gnr;
115 const o::Real radius = gr*(ring_id+1)/gnr;
116 const o::Real deg = gt + (((o::Real)point_id)/gppr * 360);
117 const o::Real rad = deg * torad;
118 ring_points[id*2] = coords[vert_id*2] + radius * Kokkos::cos(rad);
119 ring_points[id*2+1] = coords[vert_id*2+1] + radius * Kokkos::sin(rad);
120 };
121 o::parallel_for(num_points, generateRingPoints, "generateRingPoints");
122
123 //Project ring points to +/- planes
124 o::Write<o::Real> forward_ring_points(num_points * 2);
125 o::Write<o::Real> backward_ring_points(num_points * 2);
126 auto projectCoords = OMEGA_H_LAMBDA(const o::LO& id) {
127 //TODO Project the points along field lines
128 for (int i = 0; i < 2; ++i) {
129 forward_ring_points[id*2+i] = ring_points[id*2+i];
130 backward_ring_points[id*2+i] = ring_points[id*2+i];
131 }
132 };
133 o::parallel_for(num_points, projectCoords, "projectCoords");
134
135 //Use adjacency search to find the element that the projected point is in
136 o::Write<o::LO> starting_element(num_points,0, "starting_element");
137 auto verts2Elm = mesh->ask_up(0, mesh->dim());
138 auto setInitialElement = OMEGA_H_LAMBDA(const o::LO& id) {
139 //Note: Setting initial element to be the first element adjacent to the vertex
140 const o::LO vert_id = id / gppr / gnr;
141 const auto firstElm = verts2Elm.a2ab[vert_id];
142 starting_element[id] = verts2Elm.ab2b[firstElm];
143 };
144 o::parallel_for(num_points, setInitialElement, "setInitialElement");
145
146 //Calculate centroids of each element
147 o::Write<o::Real> element_centroids(mesh->nelems()*2);
148 auto cells2nodes = mesh->get_adj(mesh->dim(), o::VERT).ab2b;
149 auto calculateCentroids = OMEGA_H_LAMBDA(const o::LO& elm) {
150 auto elmVerts = o::gather_verts<3>(cells2nodes, elm);
151 auto vtxCoords = o::gather_vectors<3,2>(coords, elmVerts);
152 auto center = average(vtxCoords);
153 element_centroids[elm*2] = center[0];
154 element_centroids[elm*2+1] = center[1];
155 };
156 o::parallel_for(mesh->nelems(), calculateCentroids, "calculateCentroids");
157
158 //Create both mapping
159 forward_map = searchAndBuildMap(mesh, o::Reals(element_centroids),
160 o::Reals(forward_ring_points),
161 o::LOs(starting_element));
162 backward_map = searchAndBuildMap(mesh, o::Reals(element_centroids),
163 o::Reals(backward_ring_points),
164 o::LOs(starting_element));
165 Kokkos::Profiling::popRegion();
166}

168void gyroScatter(o::Mesh* mesh, PS* ptcls, o::LOs v2v, std::string scatterTagName) {
169 const auto btime = pumipic_prebarrier(MPI_COMM_WORLD);
170 Kokkos::Timer timer;
171 Kokkos::Profiling::pushRegion("xgcm_gyroScatter");
172 int rank, comm_size;
173 MPI_Comm_rank(MPI_COMM_WORLD,&rank);
174 MPI_Comm_size(MPI_COMM_WORLD,&comm_size);
175 const auto gr = gyro_rmax;
176 const auto gnr = gyro_num_rings;
177 const auto gppr = gyro_points_per_ring;
178 auto elm2verts = mesh->ask_down(mesh->dim(), 0);
179 auto nvpe = 3; //triangles
180 const double ringWidth = gr/gnr;
181 o::Write<o::Real> ring_accum(gnr*mesh->nverts(),0, "ring_accumulator");
182 auto accumulateToRings = PS_LAMBDA(const int& e, const int& pid, const int& mask) {
183 if(mask > 0) {
184 const auto ptclRadius = ringWidth*1.125; //TODO compute the radius
185 assert(ptclRadius >= ringWidth);
186 auto ringDown = 0;
187 for(int i=2; i<=gnr; i++)
188 ringDown += (ptclRadius >= ringWidth*i);
189 auto ringUp = ringDown+1;
190 assert(ringUp<gnr);
191 assert(ptclRadius >= ringWidth*(ringDown+1));
192 assert(ptclRadius < ringWidth*(ringUp+1));
193 const auto firstVtx = e*nvpe;
194 for(int i=0; i<nvpe; i++) {
195 const auto v = elm2verts.ab2b[firstVtx+i];
196 const auto vtxIdx = v*gnr;
197 const auto ringUpIdx = vtxIdx+ringUp;
198 const auto ringDownIdx = vtxIdx+ringDown;
199 Kokkos::atomic_fetch_add(&(ring_accum[ringUpIdx]), 1); //replace with add
200 Kokkos::atomic_fetch_add(&(ring_accum[ringDownIdx]), 1); //replace with add
201 }
202 }
203 };
204 ps::parallel_for(ptcls,accumulateToRings);
205 const Omega_h::LO nverts = mesh->nverts();
206 o::Write<o::Real> scatter_w(mesh->nverts(),0,"scatterTag_w");
207 auto scatterToMappedVerts = OMEGA_H_LAMBDA(const o::LO& v) {
208 const auto vtxIdx = v*gnr*gppr;
209 const auto gyroVtxIdx = v*gnr;
210 for(int ring=0; ring<gnr; ring++) {
211 const auto accumRingVal = ring_accum[gyroVtxIdx+ring]/gppr;
212 const auto ringIdx = ring*gppr;
213 for(int pt=0; pt<gppr; pt++) {
214 const auto ptIdx = 3*(vtxIdx+ringIdx+pt);
215 for(int elmVtx=0; elmVtx<nvpe; elmVtx++) {
216 const auto mappedIdx = ptIdx + elmVtx;
217 const auto mappedVtx = v2v[mappedIdx];
218 if (mappedVtx >= 0)
219 Kokkos::atomic_fetch_add(&(scatter_w[mappedVtx]), accumRingVal); //replace with add
220 }
221 }
222 }
223 };
224 o::parallel_for(mesh->nverts(), scatterToMappedVerts, "xgcm_scatterToMappedVerts");
225 mesh->set_tag(o::VERT, scatterTagName, o::Reals(scatter_w));
226
227 pumipic::RecordTime("gyro scatter", timer.seconds(), btime);
228 Kokkos::Profiling::popRegion();
229}

231void gyroSync(p::Mesh& picparts, const std::string& fwdTagName,
232 const std::string& bkwdTagName, const std::string& syncTagName) {
233 const auto btime = pumipic_prebarrier(MPI_COMM_WORLD);
234 Kokkos::Timer timer;
235 Kokkos::Profiling::pushRegion("xgcm_gyroSync");
236 int rank, comm_size;
237 MPI_Comm_rank(MPI_COMM_WORLD,&rank);
238 MPI_Comm_size(MPI_COMM_WORLD,&comm_size);
239 Omega_h::Write<Omega_h::Real> sync_array = picparts.createCommArray(0, 2, Omega_h::Real(0.0));
240 Omega_h::Mesh* mesh = picparts.mesh();
241 Omega_h::Read<Omega_h::Real> fwdTag = mesh->get_array<Omega_h::Real>(0, fwdTagName);
242 Omega_h::Read<Omega_h::Real> bkwdTag = mesh->get_array<Omega_h::Real>(0, bkwdTagName);
243
244 auto setSyncArray = OMEGA_H_LAMBDA(const Omega_h::LO vtx_id) {
245 sync_array[2*vtx_id] = fwdTag[vtx_id];
246 sync_array[2*vtx_id+1] = bkwdTag[vtx_id];
247 };
248 Omega_h::parallel_for(mesh->nverts(), setSyncArray);
249
250 Kokkos::Timer reducetimer;
251 picparts.reduceCommArray(0, p::Mesh::Op::SUM_OP, sync_array);
252 const auto rtime = reducetimer.seconds();
253
254 mesh->set_tag(0, syncTagName, Omega_h::Reals(sync_array));
255 pumipic::RecordTime("gyro sync", timer.seconds(), btime);
256 pumipic::RecordTime("gyro reduction", rtime);
257 Kokkos::Profiling::popRegion();
