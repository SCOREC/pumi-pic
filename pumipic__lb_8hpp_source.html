129 void ParticleBalancer::addWeights(Mesh& picparts, PS* ptcls,
130 typename PS::kkLidView new_elems,
131 typename PS::kkLidView new_procs) {
132 MPI_Comm comm = picparts.comm()->get_impl();
133 int comm_rank = picparts.comm()->rank();
134 // Device map of number of particles already assigned to another process
135 Kokkos::UnorderedMap<int, agi::wgt_t> forcedPtcls(picparts.numBuffers(picparts->dim()));
136 Omega_h::Write<Omega_h::LO> buffered_ranks(picparts.bufferedRanks(picparts->dim()));
137 Kokkos::parallel_for(buffered_ranks.size(), KOKKOS_LAMBDA(const int i) {
138 forcedPtcls.insert(buffered_ranks[i], 0);
139 });
140
141 //Count particles in each sbar & count particles already being migrated
142 Omega_h::Write<agi::wgt_t> weights(sbar_ids.size() + 1, 0);
143 Omega_h::LOs elem_sbars = getSbarIDs(picparts);
144 auto sbar_to_vert_local = sbar_to_vert;
145 auto accumulateWeight = PS_LAMBDA(const int elm, const int ptcl, const bool mask) {
146 if (mask) {
147 const int new_rank = new_procs(ptcl);
148 if (new_rank == comm_rank) {
149 const int e = new_elems(ptcl);
150 if (e != -1) {
151 int sbar_index = elem_sbars[e];
152 if (sbar_to_vert_local.exists(sbar_index)) {
153 auto index = sbar_to_vert_local.find(sbar_index);
154 const agi::lid_t vert_index = sbar_to_vert_local.value_at(index);
155 Kokkos::atomic_add(&(weights[vert_index]), 1.0);
156 }
157 }
158 }
159 else {
160 const auto index = forcedPtcls.find(new_rank);
161 Kokkos::atomic_add(&(forcedPtcls.value_at(index)), 1.0);
162 }
163 }
164 };
165 parallel_for(ptcls, accumulateWeight, "accumulateWeight");
166
167 //Transfer map to host
168 Omega_h::Write<int> owners(buffered_ranks.size(), "owners");
169 Omega_h::Write<agi::wgt_t> wgts(buffered_ranks.size(), "owners");
170 Omega_h::Write<int> index(1, 0);
171 Kokkos::parallel_for(forcedPtcls.capacity(), KOKKOS_LAMBDA (uint32_t i) {
172 if( forcedPtcls.valid_at(i) ) {
173 const int map_index = Kokkos::atomic_fetch_add(&(index[0]), 1);
174 owners[map_index] = forcedPtcls.key_at(i);
175 wgts[map_index] = forcedPtcls.value_at(i);
176 }
177 });
178 Omega_h::HostWrite<int> owners_host(owners);
179 Omega_h::HostWrite<agi::wgt_t> wgts_host(wgts);
180
181 //Send wgts to peers
182 int num_peers = owners_host.size();
183 agi::wgt_t* peer_wgts = new agi::wgt_t[num_peers];
184 MPI_Request* send_requests = new MPI_Request[num_peers];
185 MPI_Request* recv_requests = new MPI_Request[num_peers];
186 for (int i = 0; i < num_peers; ++i) {
187 MPI_Irecv(peer_wgts + i, 1, MPI_DOUBLE, owners_host[i],
188 0, comm, recv_requests + i);
189 MPI_Isend(&(wgts_host[i]), 1, MPI_DOUBLE, owners_host[i],
190 0, comm, send_requests + i);
191 }
192 MPI_Waitall(num_peers, recv_requests, MPI_STATUSES_IGNORE);
193 delete [] recv_requests;
194
195 //Accumulate all received weight on the last vertex
196 Omega_h::HostWrite<Omega_h::Real> weights_host(weights);
197 for (int i = 0; i < num_peers; ++i) {
198 weights_host[weights_host.size() - 1] += peer_wgts[i];
199 }
200
201 weightGraph->setWeights(weights_host.data());
202 MPI_Waitall(num_peers, send_requests, MPI_STATUSES_IGNORE);
203 delete [] send_requests;
204 delete [] peer_wgts;
205 }

289 template <class ViewT>
290 Kokkos::View<int*> ParticleBalancer::selectParticles(Mesh& picparts, ViewT ptcls_per_elem, ParticlePlan plan, int selection_iterations) {
291 int comm_size = picparts.comm()->size();
292 if (comm_size == 1)
293 return Kokkos::View<int*>(0);
294 int comm_rank = picparts.comm()->rank();
295
296 //Offset sum the ptcls per elem and get total number of particles
297 ViewT offsets = ViewT("ptcl per elem offsets", ptcls_per_elem.size());
298 exclusive_scan(ptcls_per_elem, offsets, Kokkos::DefaultExecutionSpace());
299
300 //Array of new processes per particle
301 lid_t nptcls = getLastValue(offsets) + getLastValue(ptcls_per_elem);
302 Kokkos::View<int*> new_procs("new procs", nptcls);
303 auto setSelf = OMEGA_H_LAMBDA(const int ptcl) {
304 new_procs[ptcl] = comm_rank;
305 };
306 Omega_h::parallel_for(new_procs.size(), setSelf, "setSelf");
307
308
309 //Select particles to migrate
310 Omega_h::LOs sbars = getSbarIDs(picparts);
311 auto send_wgts = plan.send_wgts;
312 auto sbar_to_index = plan.sbar_to_index;
313 auto part_ids = plan.part_ids;
314 auto owners = picparts.entOwners(picparts->dim());
315 for (int i = 0; i < selection_iterations; ++i) {
316 auto selectParticles = OMEGA_H_LAMBDA(const int elm) {
317 const Omega_h::LO sbar = sbars[elm];
318 const Omega_h::LO start_ptcl = offsets[elm];
319 if (sbar_to_index.exists(sbar)) {
320 const auto map_index = sbar_to_index.find(sbar);
321 for (Omega_h::LO i = 0; i < ptcls_per_elem[elm]; ++i) {
322 const Omega_h::LO new_part = new_procs[start_ptcl+i];
323 if (new_part == comm_rank) {
324 const Omega_h::LO index = sbar_to_index.value_at(map_index);
325 const Omega_h::LO part = part_ids[index];
326 const Omega_h::Real wgt = Kokkos::atomic_fetch_add(&(send_wgts[index]), -1);
327 if (part >= 0) {
328 if (wgt == 0)
329 Kokkos::atomic_add(&(sbar_to_index.value_at(map_index)), 1);
330 if (wgt > 0)
331 new_procs[start_ptcl+i] = part;
332 }
333 }
334 }
335 }
336 };
337 Omega_h::parallel_for(ptcls_per_elem.size(), selectParticles, "selectParticles");
338 Omega_h::Write<Omega_h::LO> finished(1,0);
339 auto checkPlan = OMEGA_H_LAMBDA(const int id) {
340 if (send_wgts[id] > 0) {
341 finished[0] = 1;
342 }
343 };
344 Omega_h::parallel_for(send_wgts.size(), checkPlan);
345 Omega_h::HostWrite<Omega_h::LO> finished_h(finished);
346 if (finished_h[0] == 0)
347 break;
348 }
349 return new_procs;
350 }

382 int np = ptcls->nPtcls();
383 int min_p, max_p, tot_p;
384 MPI_Reduce(&np, &min_p, 1, MPI_INT, MPI_MIN, 0, comm);
385 MPI_Reduce(&np, &max_p, 1, MPI_INT, MPI_MAX, 0, comm);
386 MPI_Reduce(&np, &tot_p, 1, MPI_INT, MPI_SUM, 0, comm);
387
388 int comm_rank;
389 MPI_Comm_rank(comm, &comm_rank);
390 int comm_size;
391 MPI_Comm_size(comm, &comm_size);
392 if (comm_rank == 0) {
393 float avg = tot_p / comm_size;
394 float imb = max_p / avg;
395 printInfo("Ptcl LB <max, min, avg, imb>: %d %d %.3f %.3f\n", max_p, min_p, avg, imb);
