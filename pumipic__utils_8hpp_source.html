94template <class T> OMEGA_H_DEVICE
95Omega_h::LO find_min_no_exp(const T a, const T exp_num, Omega_h::LO n){
96
97 Omega_h::LO ind=0;
98 Omega_h::Real min = a[0];
99 for (Omega_h::LO i=0; i<n; ++i){
100 if(exp_num[i]==0){
101 ind = i;
102 min = a[i];
103 break;
104 }
105 }
106
107
108 Omega_h::LO ind_track=0;
109 Omega_h::LO ind_ret=0;
110
111 for(Omega_h::LO i=ind; i<n-1; ++i) {
112 if(exp_num[i+1]==0) {
113 ind_track++;
114 if(a[i+1]<min){
115 min = a[i+1];
116 ind = i+1;
117 ind_ret=ind_track;
118 }
119
120 }
121 }
122 return ind_ret;
123}

179 const o::Real &z, o::Real &r, o::Real &theta, o::Real &phi) {
180 r = Kokkos::sqrt(x*x + y*y + z*z);

186OMEGA_H_DEVICE o::Real interpolate2dField(const o::Reals& data,
187 const o::Real gridx0, const o::Real gridz0, const o::Real dx,
188 const o::Real dz, const o::LO nx, const o::LO nz,
189 const o::Vector<3> &pos, const bool cylSymm = true,
190 const o::LO nComp = 1, const o::LO comp = 0, bool debug= false) {
191
192 if(nx*nz == 1) {
193 return data[comp];
194 }
195 o::Real fxz = 0;
196 o::Real fx_z1 = 0;
197 o::Real fx_z2 = 0;
198 auto dim1 = pos[0];
199 auto z = pos[2];
200 if(cylSymm) {
201 dim1 = Kokkos::sqrt(pos[0]*pos[0] + pos[1]*pos[1]);
202 }
203
204 OMEGA_H_CHECK(dx >0 && dz>0);
205 o::LO i = Kokkos::floor((dim1 - gridx0)/dx);
206 o::LO j = Kokkos::floor((z - gridz0)/dz);
207
208 if (i < 0) i=0;
209 if (j < 0) j=0;
210
211 auto gridXi = gridx0 + i * dx;
212 auto gridXip1 = gridx0 + (i+1) * dx;
213 auto gridZj = gridz0 + j * dz;
214 auto gridZjp1 = gridz0 + (j+1) * dz;
215
216 if (i >=nx-1 && j>=nz-1) {
217 fxz = data[(nx-1+(nz-1)*nx)*nComp+comp];
218 }
219 else if (i >=nx-1) {
220 fx_z1 = data[(nx-1+j*nx)*nComp + comp];
221 fx_z2 = data[(nx-1+(j+1)*nx)*nComp + comp];
222 fxz = ((gridZjp1-z)*fx_z1+(z - gridZj)*fx_z2)/dz;
223 }
224 else if (j >=nz-1) {
225 fx_z1 = data[(i+(nz-1)*nx)*nComp + comp];
226 fx_z2 = data[(i+(nz-1)*nx)*nComp+ comp];
227 fxz = ((gridXip1-dim1)*fx_z1+(dim1 - gridXi)*fx_z2)/dx;
228 }
229 else {
230 fx_z1 = ((gridXip1-dim1)*data[(i+j*nx)*nComp + comp] +
231 (dim1 - gridXi)*data[(i+1+j*nx)*nComp + comp])/dx;
232 fx_z2 = ((gridXip1-dim1)*data[(i+(j+1)*nx)*nComp + comp] +
233 (dim1 - gridXi)*data[(i+1+(j+1)*nx)*nComp + comp])/dx;
234 fxz = ((gridZjp1-z)*fx_z1+(z - gridZj)*fx_z2)/dz;
235
236 }
237 if(debug)
238 printInfo("interp2dField pos: %g %g %g : dim1 %g nx %d nz %d gridx0 %g "
239 "gridz0 %g grid1 %g grid2 %g i %d j %d dx %g dz %g fxz %g \n",
240 pos[0], pos[1], pos[2], dim1, nx, nz, gridx0, gridz0, gridXi,
241 gridXip1, i, j, dx, dz, fxz);
242 return fxz;
243}

260OMEGA_H_DEVICE o::Real interpolate2d(const o::Reals& data, const o::Real gridXi,
261 const o::Real gridXip1, const o::Real gridZj, const o::Real gridZjp1,
262 const o::Real x0, const o::Real z, const o::LO nx, const o::LO nz, const int i,
263 const int j, const o::Real dx, const o::Real dz, const o::Real y=0,
264 const bool cylSymm = true, const o::LO nComp = 1, const o::LO comp = 0, bool debug=false) {
265 if(nx <= 1 && nz <= 1)
266 return data[comp];
267 auto x = x0;
268 if(cylSymm)
269 x = Kokkos::sqrt(x*x + y*y);
270 o::Real fxz = 0;
271 if (i >=nx-1 && j>=nz-1) {
272 fxz = data[(nx-1+(nz-1)*nx)*nComp+comp];
273 }
274 else if (i >=nx-1) {
275 fxz = interpolate2d_based(data, (nx-1+j*nx)*nComp + comp,
276 (nx-1+(j+1)*nx)*nComp + comp, z - gridZj, gridZjp1-z, z, dz);
277 }
278 else if (j >=nz-1) {
279 fxz = interpolate2d_based(data, (i+(nz-1)*nx)*nComp + comp,
280 (i+(nz-1)*nx)*nComp+ comp, x - gridXi, gridXip1-x, x, dx);
281 }
282 else {
283 auto fx_z1 = interpolate2d_based(data, (i+j*nx)*nComp + comp,
284 (i+1+j*nx)*nComp + comp, gridXi, gridXip1, x, dx);
285 auto fx_z2 = interpolate2d_based(data, (i+(j+1)*nx)*nComp + comp,
286 (i+1+(j+1)*nx)*nComp + comp, gridXi, gridXip1, x, dx);
287 fxz = interpolate2d_base(fx_z1, fx_z2, gridZj, gridZjp1, z, dz);
288 if(debug)
289 printInfo("fx_z1 %.15f fx_z2 %.15f \n", fx_z1, fx_z2);
290 }
291 if(debug)
292 printInfo("int2d: pos %g %g %g dx,dz %g %g i,j %d %d nx,nz %d %d \n"
293 " grids %.15f %.15f %.15f %.15f fxz %.15f\n\n", x,y,z, dx, dz, i, j, nx, nz,
294 gridXi, gridXip1, gridZj, gridZjp1, fxz);
295 return fxz;
296}

298OMEGA_H_DEVICE o::Real interpolate2d_field(const o::Reals& data, const o::Real gridx0,
299 const o::Real gridz0, const o::Real dx, const o::Real dz, const o::LO nx,
300 const o::LO nz, const o::Vector<3> &pos, const bool cylSymm = true,
301 const o::LO nComp = 1, const o::LO comp = 0, bool debug= false) {
302 if(nx <=1 && nz <= 1)
303 return data[comp];
304 auto x = pos[0];
305 auto z = pos[2];
306 if(cylSymm)
307 x = Kokkos::sqrt(x*x+pos[1]*pos[1]);
308 OMEGA_H_CHECK(dx >0 && dz>0);
309 o::LO i = Kokkos::floor((x - gridx0)/dx);
310 o::LO j = Kokkos::floor((z - gridz0)/dz);
311 if (i < 0) i=0;
312 if (j < 0) j=0;
313 auto gridXi = gridx0 + i * dx;
314 auto gridXip1 = gridx0 + (i+1) * dx;
315 auto gridZj = gridz0 + j * dz;
316 auto gridZjp1 = gridz0 + (j+1) * dz;
317 if(debug)
318 printInfo("2d_field: pos %g %g %g dx,z %g %g i,j %d %d grids %g %g %g %g\n",
319 x,pos[1],z, dx, dz, i, j, gridXi, gridXip1, gridZj, gridZjp1);
320 return interpolate2d(data, gridXi, gridXip1, gridZj, gridZjp1, x, z, nx,
321 nz, i, j, dx, dz, 0, false, nComp, comp, debug);
322}

333OMEGA_H_DEVICE o::Real interpolate2d_wgrid(const o::Reals& data,
334 const o::Reals& gridx, const o::Reals& gridz, const o::Vector<3>& pos,
335 const bool cylSymm = true, const o::LO nComp = 1, const o::LO comp = 0,
336 bool debug = false) {
337 int nx = gridx.size();
338 int nz = gridz.size();
339 if(debug)
340 printInfo("nx %d nz %d comp %d nComp %d \n", nx, nz, comp, nComp);
341 if(nx <=1 || nz <= 1)
342 return data[comp];
343 auto x = pos[0];
344 auto z = pos[2];
345 x = (cylSymm) ? Kokkos::sqrt(x*x+pos[1]*pos[1]) : x;
346 auto dx = gridx[1] - gridx[0]; /*(gridx[nx-1] - gridx[0])/nx*/
347 auto dz = gridz[1] - gridz[0]; /*(gridz[nz-1] - gridz[0])/nz*/
348 OMEGA_H_CHECK(dx >0 && dz>0);
349 o::LO i = Kokkos::floor((x - gridx[0])/dx);
350 o::LO j = Kokkos::floor((z - gridz[0])/dz);
351 i = (i < 0) ? 0 : i;
352 j = (j < 0) ? 0 : j;
353 //off limit values not used in calculation
354 auto gridXi = (i >= nx) ? gridx[nx-1] : gridx[i]; /*gridx0 + i * dx*/
355 auto gridXip1 = (i>= nx-1) ? gridx[nx-1] : gridx[i+1]; /*gridx0 + (i+1) * dx*/
356 auto gridZj = (j>=nz) ? gridz[nz-1] : gridz[j]; /*gridz0 + j * dz */
357 auto gridZjp1 = (j>=nz-1) ? gridz[nz-1] : gridz[j+1]; /*gridz0 + (j+1) * dz*/
358 if(debug)
359 printInfo("pos %g %g %g dx,z %g %g i,j %d %d grids %g %g %g %g\n",
360 x,pos[1],z, dx, dz, i, j, gridXi, gridXip1, gridZj, gridZjp1);
361 return interpolate2d(data, gridXi, gridXip1, gridZj, gridZjp1, x, z, nx,
362 nz, i, j, dx, dz, 0, false, nComp, comp, debug);
363}

377OMEGA_H_DEVICE o::Real interpolate3d_field(const o::Real x, const o::Real y,
378 const o::Real z, int nx, int ny, int nz, const o::Reals& gridx,
379 const o::Reals& gridy, const o::Reals& gridz, const o::Reals& data) {
380 /*auto nx = gridx.size();
381 auto ny = gridy.size();
382 auto nz = gridz.size(); */
383 bool debug = false;
384 if(debug)
385 for(int i=0; i<5; ++i) printInfo(" %d %.15e \n", i, gridz[i]);
386 o::Real fxyz = 0;
387 o::Real dx = gridx[1] - gridx[0];
388 o::Real dy = gridy[1] - gridy[0];
389 o::Real dz = gridz[1] - gridz[0];
390 OMEGA_H_CHECK(!(o::are_close(dx, 0) || o::are_close(dy, 0) || o::are_close(dz, 0)));
391 int i = Kokkos::floor((x - gridx[0])/dx);
392 int j = Kokkos::floor((y - gridy[0])/dy);
393 int k = Kokkos::floor((z - gridz[0])/dz);
394 i = (i < 0) ? 0 : ((i >= nx-1) ? (nx-2) : i);
395 j = (j < 0 || ny <= 1) ? 0 : ((j >= ny-1) ? (ny-2) : j);
396 k = (k < 0 || nz <= 1) ? 0 : ((k >= nz-1) ? (nz-2) : k);
397
398 auto fx_z0 = interpolate2d_baseg(data, i + j*nx + k*nx*ny, gridx, i, x, dx);
399 auto fx_z1 = interpolate2d_baseg(data, i + j*nx + (k+1)*nx*ny,
400 gridx, i, x, dx);
401 auto fxy_z0 = interpolate2d_baseg(data, i + (j+1)*nx + k*nx*ny,
402 gridx, i, x, dx);
403 auto fxy_z1 = interpolate2d_baseg(data,i + (j+1)*nx + (k+1)*nx*ny,
404 gridx, i, x, dx);
405 auto fxz0 = interpolate2d_base(fx_z0, fx_z1, gridz[k], gridz[k+1], z, dz);
406 auto fxz1 = interpolate2d_base(fxy_z0, fxy_z1, gridz[k], gridz[k+1], z, dz);
407 fxyz = interpolate2d_base(fxz0, fxz1, gridy[j], gridy[j+1], y, dy);
408 if(debug) {
409 printInfo("fx_z0 %.15e fx_z1 %.15e \n", fx_z0, fx_z1);
410 printInfo("fxy_z0 %.15e fxy_z1 %.15e fxz0 %.15e fxz1 %.15e fxyz %.15e\n",
411 fxy_z0, fxy_z1, fxz0, fxz1, fxyz);
412 printInfo("x %.15e y %.15e z %.15e i %d j %d k %d dx %.15e dy %.15e dz %.15e \n",
413 x, y, z, i, j, k, dx, dy, dz);
414 }
415 fxyz = (ny <= 1) ? fxz0: fxyz;
416 fxyz = (nz <= 1) ? fx_z0: fxyz;
417 if(debug)
418 printInfo("fxy %.15e\n", fxyz);
419 return fxyz;
420}

600OMEGA_H_DEVICE o::LO is_face_within_limit_from_tet(const o::Matrix<3, 4>& tet,
601 const Omega_h::LOs& face_verts, const Omega_h::Reals& coords, const o::LO face_id,
602 const o::Real depth) {
603 auto fv2v = Omega_h::gather_verts<3>(face_verts, face_id); //Few<LO, 3>
604 const auto face = Omega_h::gather_vectors<3, 3>(coords, fv2v);
605 o::LO res = 0;
606 for(o::LO i=0; i<3; ++i) { //3 vtx of face
607 for(o::LO j=0; j<4; ++j) { //4 vtx of tet
608 auto dv = face[i] - tet[j];
609 auto d2 = o::norm(dv);
610 if(d2 <= depth) {
611 res = 1;
612 break;
613 }
614 if(res)
615 break;
616 }
617 }
618 return res;
619}
